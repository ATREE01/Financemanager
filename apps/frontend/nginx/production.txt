server {
    listen 80;
    listen [::]:80;

    server_name financemanager.ddns.net;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https:/financemanager.ddns.net$request_uri;
    }
}

server {
    listen 443 ssl;
    listen [::]:443 ssl;

    server_name atree.ddns.net;

    ssl_certificate /etc/letsencrypt/live/financemanager.ddns.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/financemanager.ddns.net/privkey.pem;

    location / {
        root   /usr/share/nginx/html;
        index  index.html index.htm;
        try_files $uri $uri/ /index.html;
    }

    location /api {
        proxy_pass http://backend:3000/api;
    }
}
